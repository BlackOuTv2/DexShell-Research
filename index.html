Python — full header reconstruction
import struct, hashlib, zlib

SIZE_OFF_PAIRS = [
    (0x38, 0x3c),   # string_ids_size → string_ids_off
    (0x40, 0x44),   # type_ids_size   → type_ids_off
    (0x48, 0x4c),   # proto_ids_size  → proto_ids_off
    (0x50, 0x54),   # field_ids_size  → field_ids_off
    (0x58, 0x5c),   # method_ids_size → method_ids_off
    (0x60, 0x64),   # class_defs_size → class_defs_off
    (0x68, 0x6c),   # data_size       → data_off
]
ENTRY_SIZES = {0x3c:4, 0x44:4, 0x4c:12, 0x54:8, 0x5c:8, 0x64:32, 0x6c:1}

def deobfuscate_dex(data):
    d = bytearray(data)
    d[0:8] = b'dex\n035\x00'        # restore magic
    off = 0x70                       # start after fixed header
    for sz_off, id_off in SIZE_OFF_PAIRS:
        count = struct.unpack_from('<I', d, sz_off)[0]
        if count > 0:
            struct.pack_into('<I', d, id_off, off)
            off += count * ENTRY_SIZES[id_off]
            off = (off + 3) & ~3     # 4-byte align
    struct.pack_into('<I', d, 32, len(d))  # fix file_size
    # recompute adler32 + sha1 ...
    return bytes(d)
